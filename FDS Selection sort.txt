#include<stdio.h>
void main()
{
    int a[10],m,i,j,n,temp;
    printf("Enter the size of an array:");
    scanf("%d",&m);
    printf("Enter the elements of an array: \n");
    for(i=0;i<m;i++)
    {
        printf("a[%d] = ",i);
        scanf("%d",&a[i]);
    }
    for(i=1;i<m;i++)
    {
        for(j=0;j<i;j++)
        {
            if(a[i]<a[j])
            {
                temp=a[j];
                a[j]=a[i];
                for(n=i;n>j;n--)
                {
                    a[n]=a[n-1];
                }
                a[n+1]=temp;
            }
        }
    }
    printf("\nsorted array is: ");
    for(i=0;i<m;i++)
    {
        printf("\n%d",a[i]);
    }
}
